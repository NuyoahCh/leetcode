```go
func isOneBitCharacter(bits []int) bool {
	n := len(bits)
	i := 0
	for i < n-1 { // 循环直到剩下至多一个数字
		i += bits[i] + 1 // 如果 bits[i] == 1 则跳过下一位
	}
	return i == n-1 // 注意题目保证 bits[n-1] == 0，无需判断
}
```
