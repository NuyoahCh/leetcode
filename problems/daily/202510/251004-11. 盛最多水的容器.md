func maxArea(height []int) int {
	left, right := 0, len(height) - 1
	res := 0
	for left <= right {
		max_area := min(height[left], height[right]) * (right - left)
		if height[left] < height[right] {
			left++
		} else {
			right--
		}
		res = max(res, max_area)
	}
	return res
}
