func findLexSmallestString(s string, a int, b int) string {
	n := len(s)
	vis := make([]bool, n)
	res := s
	// 将 s 延长一倍，方便截取轮转后的字符串 t
	s = s + s
	for i := 0; !vis[i]; i = (i + b) % n {
		vis[i] = true
		for j := 0; j < 10; j++ {
			kLimit := 0
			if b % 2 != 0 {
				kLimit = 9
			}
			for k := 0; k <= kLimit; k++ {
				// 每次进行累加之前，重新截取 t
				t := []byte(s[i : i+n])
				for p := 1; p < n; p += 2 {
					t[p] = '0' + byte((int(t[p] - '0') + j * a) % 10)
				}
				for p := 0; p < n; p += 2 {
					t[p] = '0' + byte((int(t[p] - '0') + k * a) % 10)
				}
				tStr := string(t)
				if tStr < res {
					res = tStr
				}
			}
		}
	}
	return res
}
