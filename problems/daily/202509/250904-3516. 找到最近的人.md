func findClosest(x int, y int, z int) int {
    dxz := int(math.Abs(float64(x - z)))
    dyz := int(math.Abs(float64(y - z)))
    if dxz < dyz {
        return 1
    } else if dxz > dyz {
        return 2
    } else {
        return 0
    }
}
