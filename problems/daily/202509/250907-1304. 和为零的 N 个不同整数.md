func sumZero(n int) []int {
    ans := make([]int, 0, n)
    for i := 1; i <= n / 2; i++ {
        ans = append(ans, i, -i)
    }
    if n % 2 == 1 {
        ans = append(ans, 0)
    }
    return ans
}
