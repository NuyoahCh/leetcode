func maxFreqSum(s string) int {
    mp := make(map[byte]int)
    for i := 0; i < len(s); i++ {
        mp[s[i]]++
    }
    
    vowel, consonant := 0, 0
    for ch := 'a'; ch <= 'z'; ch++ {
        count := mp[byte(ch)]
        if isVowel(byte(ch)) {
            vowel = max(vowel, count)
        } else {
            consonant = max(consonant, count)
        }
    }
    return vowel + consonant
}

func isVowel(c byte) bool {
    return c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u'
}
