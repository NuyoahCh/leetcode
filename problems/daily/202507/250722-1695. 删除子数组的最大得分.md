func maximumUniqueSubarray(nums []int) int {
    n := len(nums)
    psum := make([]int, n + 1)
    cnt := make(map[int]int)
    ans, pre := 0, 0
    for i := 0; i < n; i++ {
        psum[i + 1] = psum[i] + nums[i]
        if val, exists := cnt[nums[i]]; exists {
            pre = max(pre, val)
        }
        ans = max(ans, psum[i + 1] - psum[pre]);
        cnt[nums[i]] = i + 1
    }
    return ans
}
