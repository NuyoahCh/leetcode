func zeroFilledSubarray(nums []int) (ans int64) {
	last := -1
	for i, x := range nums {
		if x != 0 {
			last = i // 记录上一个非 0 元素的位置
		} else {
			ans += int64(i - last)
		}
	}
	return
}
