/**
 * Definition for a binary tree node.
 * type TreeNode struct {
 *     Val int
 *     Left *TreeNode
 *     Right *TreeNode
 * }
 */
func maxLevelSum(root *TreeNode) int {
    if root == nil {
        return 0
    }
    q := []*TreeNode{root}
    // 记录 BFS 走到的层数
    depth := 1
    // 记录元素和最大的那一行和最大元素和
    res := 0
    maxSum := ^int(^uint(0) >> 1)

    for len(q) > 0 {
        sz := len(q)
        levelSum := 0
        // 遍历这一层
        for i := 0; i < sz; i++ {
            cur := q[0]
            q = q[1:]
            levelSum += cur.Val

            if cur.Left != nil {
                q = append(q, cur.Left)
            }
            if cur.Right != nil {
                q = append(q, cur.Right)
            }
        }
        if levelSum > maxSum {
            // 更新最大元素和
            res = depth
            maxSum = levelSum
        }
        depth++
    }
    return res
}
